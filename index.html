import pygame import random import time # 1. 초기화 pygame.init() # 2. 화면 설정 screen_width = 800 screen_height = 600 screen = pygame.display.set_mode((screen_width, screen_height)) pygame.display.set_caption("동물 먹이 주기 게임") # 3. 색상 설정 (GRAY 추가 및 오타 수정) WHITE = (255, 255, 255) BLACK = (0, 0, 0) GRAY = (128, 128, 128) GOLD = (255, 215, 0) GREEN = (100, 255, 100) # 잎 PINK = (255, 182, 193) # 츄르 BROWN = (222, 184, 135) # 개껌 RED = (255, 100, 100) # 4. 폰트 설정 def get_font(size): try: # 윈도우: malgungothic, 맥: applesdgothicneo return pygame.font.SysFont("malgungothic, applesdgothicneo, nanumgothic", size) except: return pygame.font.SysFont("arial", size) font = get_font(40) small_font = get_font(25) # 5. 게임 변수 animals = ["코알라", "고양이", "강아지"] foods = {"코알라": "잎", "고양이": "츄르", "강아지": "개껌"} score = 0 current_animal = random.choice(animals) # 먹이 버튼 설정 leaf_rect = pygame.Rect(150, 400, 150, 60) churu_rect = pygame.Rect(325, 400, 150, 60) bone_rect = pygame.Rect(500, 400, 150, 60) # 사료 버튼 변수 special_food_active = False special_food_timer = 0 special_food_rect = pygame.Rect(350, 500, 100, 50) def draw_text(text, x, y, color=BLACK, center=False): img = font.render(text, True, color) if center: text_rect = img.get_rect(center=(x, y)) screen.blit(img, text_rect) else: screen.blit(img, (x, y)) def draw_button(rect, text, color): mouse_pos = pygame.mouse.get_pos() display_color = color if rect.collidepoint(mouse_pos): display_color = tuple(min(255, c + 30) for c in color) pygame.draw.rect(screen, display_color, rect, border_radius=10) pygame.draw.rect(screen, BLACK, rect, 2, border_radius=10) text_img = small_font.render(text, True, BLACK) text_rect = text_img.get_rect(center=rect.center) screen.blit(text_img, text_rect) # 6. 게임 루프 running = True clock = pygame.time.Clock() while running: screen.fill(WHITE) current_time = time.time() # 이벤트 처리 for event in pygame.event.get(): if event.type == pygame.QUIT: running = False given_food = "" # 마우스 클릭 처리 if event.type == pygame.MOUSEBUTTONDOWN: if leaf_rect.collidepoint(event.pos): given_food = "잎" elif churu_rect.collidepoint(event.pos): given_food = "츄르" elif bone_rect.collidepoint(event.pos): given_food = "개껌" # 특수 버튼 클릭 if special_food_active and special_food_rect.collidepoint(event.pos): bonus = random.randint(-10, 10) score += bonus special_food_active = False # 키보드 조작 if event.type == pygame.KEYDOWN: if event.key == pygame.K_1: given_food = "잎" elif event.key == pygame.K_2: given_food = "츄르" elif event.key == pygame.K_3: given_food = "개껌" # 결과 처리 if given_food: if foods[current_animal] == given_food: score += 5 else: score -= 5 current_animal = random.choice(animals) # 사료 버튼 등장 로직 if not special_food_active and random.random() < 0.01: special_food_active = True special_food_timer = current_time # 5초 지나면 사라짐 if special_food_active and (current_time - special_food_timer > 5): special_food_active = False # 화면 그리기 draw_text(f"현재 점수: {score}", 20, 20) draw_text(f"배고픈 동물: [ {current_animal} ]", 400, 200, center=True) draw_text(f"{current_animal}에게 맞는 먹이를 주세요!", 400, 260, color=GRAY, center=True) # 버튼 그리기 draw_button(leaf_rect, "1. 잎", GREEN) draw_button(churu_rect, "2. 츄르", PINK) draw_button(bone_rect, "3. 개껌", BROWN) if special_food_active: draw_button(special_food_rect, "랜덤 사료!", GOLD) pygame.display.flip() clock.tick(60) pygame.quit()score = 0 current_animal = random.choice(animals) # 마우스가 버튼 위에 있을 때 색상 변경 (Hover 효과) mouse_pos = pygame.mouse.get_pos() display_color = color if rect.collidepoint(mouse_pos): # 조금 더 밝게 표시 (간단한 구현을 위해 20씩 더함) display_color = tuple(min(255, c + 30) for c in color) pygame.draw.rect(screen, display_color, rect, border_radius=10) pygame.draw.rect(screen, BLACK, rect, 2, border_radius=10) text_img = small_font.render(text, True, BLACK) text_rect = text_img.get_rect(center=rect.center) screen.blit(text_img, text_rect) # 마우스 클릭 처리 if event.type == pygame.MOUSEBUTTONDOWN: if leaf_rect.collidepoint(event.pos): given_food = "잎" elif churu_rect.collidepoint(event.pos): given_food = "츄르" elif bone_rect.collidepoint(event.pos): given_food = "개껌" # 특수 버튼 클릭 if special_food_active and special_food_rect.collidepoint(event.pos): bonus = random.randint(-10, 10) score += bonus special_food_active = False # 키보드 조작 if event.type == pygame.KEYDOWN: if event.key == pygame.K_1: given_food = "잎" elif event.key == pygame.K_2: given_food = "츄르" elif event.key == pygame.K_3: given_food = "개껌" # 결과 처리 (마우스나 키보드 입력이 있었을 때만) if given_food: if foods[current_animal] == given_food: score += 5 else: score -= 5 current_animal = random.choice(animals) # 사료 버튼 등장 로직 (확률 발생) if not special_food_active and random.random() < 0.01: # 확률 조금 높임 special_food_active = True special_food_timer = current_time # 5초 지나면 사라짐 if special_food_active and (current_time - special_food_timer > 5): special_food_active = False # 화면 그리기 draw_text(f"현재 점수: {score}", 20, 20) draw_text(f"배고픈 동물: [ {current_animal} ]", 400, 200, center=True) draw_text(f"{current_animal}에게 맞는 먹이를 주세여!", 400, 260, color=GRAY, center=True) # 버튼 그리기 draw_button(leaf_rect, "1. 잎", GREEN) draw_button(churu_rect, "2. 츄르", PINK) draw_button(bone_rect, "3. 개껌", BROWN) if special_food_active: draw_button(special_food_rect, "랜덤 사료!", GOLD) pygame.display.flip() # update 대신 flip 사용 권장 clock.tick(60) # 30프레임보다 60프레임이 더 부드럽습니다. pygame.quit()
