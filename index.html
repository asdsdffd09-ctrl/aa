import pygame import random import time import os # 1. 초기화 pygame.init() # 2. 화면 설정 (가상 디스플레이 대응을 위한 환경 변수 설정) # 깃허브 codespaces 등에서 실행 시 에러 방지용 if "DISPLAY" not in os.environ: os.environ["SDL_VIDEODRIVER"] = "dummy" screen_width = 800 screen_height = 600 screen = pygame.display.set_mode((screen_width, screen_height)) pygame.display.set_caption("동물 먹이 주기 게임") # 3. 색상 설정 WHITE = (255, 255, 255) BLACK = (0, 0, 0) GRAY = (128, 128, 128) GOLD = (255, 215, 0) GREEN = (100, 255, 100) PINK = (255, 182, 193) BROWN = (222, 184, 135) # 4. 폰트 설정 (시스템 폰트가 없을 경우를 대비해 기본 폰트 사용) def get_font(size): try: # 시스템에 한글 폰트가 없을 확률이 높으므로 pygame 기본 폰트 사용 시도 return pygame.font.SysFont("notosanscjkjp, malgungothic, applesdgothicneo", size) except: return pygame.font.Font(None, size) # 폰트가 없으면 기본 폰트(English only) font = get_font(40) small_font = get_font(25) # 5. 게임 변수 animals = ["Koala", "Cat", "Dog"] # 한글 폰트 깨짐 방지를 위해 영어 병기 권장 foods = {"Koala": "Leaf", "Cat": "Churu", "Dog": "Bone"} score = 0 current_animal = random.choice(animals) # 버튼 설정 leaf_rect = pygame.Rect(150, 400, 150, 60) churu_rect = pygame.Rect(325, 400, 150, 60) bone_rect = pygame.Rect(500, 400, 150, 60) special_food_active = False special_food_timer = 0 special_food_rect = pygame.Rect(350, 500, 100, 50) def draw_text(text, x, y, color=BLACK, center=False): img = font.render(text, True, color) if center: text_rect = img.get_rect(center=(x, y)) screen.blit(img, text_rect) else: screen.blit(img, (x, y)) def draw_button(rect, text, color): mouse_pos = pygame.mouse.get_pos() display_color = color if rect.collidepoint(mouse_pos): display_color = tuple(min(255, c + 30) for c in color) pygame.draw.rect(screen, display_color, rect, border_radius=10) pygame.draw.rect(screen, BLACK, rect, 2, border_radius=10) text_img = small_font.render(text, True, BLACK) text_rect = text_img.get_rect(center=rect.center) screen.blit(text_img, text_rect) # 6. 게임 루프 running = True clock = pygame.time.Clock() while running: screen.fill(WHITE) current_time = time.time() for event in pygame.event.get(): if event.type == pygame.QUIT: running = False given_food = "" if event.type == pygame.MOUSEBUTTONDOWN: if leaf_rect.collidepoint(event.pos): given_food = "Leaf" elif churu_rect.collidepoint(event.pos): given_food = "Churu" elif bone_rect.collidepoint(event.pos): given_food = "Bone" if special_food_active and special_food_rect.collidepoint(event.pos): score += random.randint(-10, 10) special_food_active = False if event.type == pygame.KEYDOWN: if event.key == pygame.K_1: given_food = "Leaf" elif event.key == pygame.K_2: given_food = "Churu" elif event.key == pygame.K_3: given_food = "Bone" if given_food: if foods[current_animal] == given_food: score += 5 else: score -= 5 current_animal = random.choice(animals) if not special_food_active and random.random() < 0.01: special_food_active = True special_food_timer = current_time if special_food_active and (current_time - special_food_timer > 5): special_food_active = False draw_text(f"Score: {score}", 20, 20) draw_text(f"Animal: [ {current_animal} ]", 400, 200, center=True) draw_text(f"Feed the {current_animal}!", 400, 260, color=GRAY, center=True) draw_button(leaf_rect, "1. Leaf", GREEN) draw_button(churu_rect, "2. Churu", PINK) draw_button(bone_rect, "3. Bone", BROWN) if special_food_active: draw_button(special_food_rect, "Random!", GOLD) pygame.display.flip() clock.tick(60) pygame.quit()
